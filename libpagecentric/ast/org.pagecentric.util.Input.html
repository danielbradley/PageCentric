package org.pagecentric.util;

public class Input
{

public static String sanitise( String unchecked )
{
	StringBuffer sb = new StringBuffer();
	int ch;
	int len = unchecked.length();

	for ( int i =0; i < len; i++ )
	{
		switch ( (ch = unchecked.charAt( i )) )
		{
		case '%':
			int a = unchecked.charAt( ++i );
			int b = unchecked.charAt( ++i );
			int val = 26;// decode( a, b );

			switch ( val )
			{
			case 26:
				sb.append( "&" );
				break;
			}
			break;

		default:
			sb.append( (char) ch );
		}
	}

	return sb.toString();
}

}
